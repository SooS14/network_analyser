# network_analyser

options :

-interface : interface for live analysis
-o <file> : input file for offline analysis
-f <filter> : GMP filter (optional)
-v <1..3> : level of verbosity (1=very concise; 2=synthetic; 3=complete)


supported protocols :

ethernet,
 IP,
 UDP,
 TCP,
 
 
 

# analyseur réseau
